#include <ESP8266HTTPClient.h>
#include <ESP8266WiFi.h>
#include <DHTesp.h>
DHTesp dht;
//]w办跑计A放驻卫惬
float humidity;
float temperature;

void setup() {
  //]wjv(Baud)
  Serial.begin(115200);

  //]wdht
  dht.setup(2, DHTesp::DHT11);
  //suwifiA斗恶JSSIDのPassword
  WiFi.begin("<SSID>", "<Password>");   //WiFi connection
  //单wifisuΘ\
  while (WiFi.status() != WL_CONNECTED) {  //Wait for the WiFI connection completion
    //Cj0.5沓su@Ω
    delay(500);
    Serial.println("Waiting for connection");
 
  }
 
}

void getDHT(){
  //单DHT11骇No啥
  delay(dht.getMinimumSamplingPeriod());
 // delay(dht.getMinimumSamplingPeriod());
  //o楞
  humidity = dht.getHumidity();
  //o放
  temperature = dht.getTemperature();
}
 
void loop() {
 //o放驻卫惬
 getDHT();
 
 //pGヘesuAOタ`
 if(WiFi.status()== WL_CONNECTED){   //Check WiFi connection status
 
   //廿HTTPClineンAノㄓsuServer
   HTTPClient http;    //Declare object of class HTTPClient

  //?args1=" + String(humidity) + "&args2=" + String(temperature)
   //suServer
   http.begin("http://192.168.43.160/" + String(humidity) + "/" + String(temperature));      //Specify request destination
   http.addHeader("Content-Type", "text/plain");  //Specify content-type header
   http.setUserAgent("a=10");
 
   int httpCode = http.POST("Message from ESP8266");   //Send the request
   String payload = http.getString();                  //Get the response payload
 
   Serial.println(httpCode);   //Print HTTP return code
   Serial.println(payload);    //Print request response payload
  
   http.end();  //Close connection
 
 }else{
 
    Serial.println("Error in WiFi connection");   
 
 }
  //C30沓su@Ω
  delay(30000);  //Send a request every 30 seconds
 
}
